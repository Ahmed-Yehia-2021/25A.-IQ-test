#include <iostream>
#include <bits/stdc++.h>
#include <string>
typedef long long ll;
using namespace std;

int main()
{
    int n;
    cin>>n;

    int arr[n], evens=0, odds=0, pos1, pos2;
    for(int x=0;x<n;x++){
        cin>>arr[x];
        if( arr[x]%2==0 ){ evens++;  pos1=x; }
        else{ odds++;  pos2=x; }
    }

    if( evens>odds ){ cout<<pos2+1; }
    else{ cout<<pos1+1; }

    return 0;
}
